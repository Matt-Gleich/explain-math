    \section{Basic Math Example}
        \begin{align*}
            \initialeq{-2(x + 2) = 6}
            \explain{Distribute out -2}{
                (-2 \cdot x) + (-2 \cdot 2) = 6 \\
                -2x - 4 = 6
            }
            \explain{Remove -4 from the left}{
                -2x - 4 + 4 = 6 + 4 \\
                -2x = 10
            }
            \explain{Remove -2 from the left}{
                \frac{-2x}{-2} = \frac{10}{-2} \\
                x = -5
            }
        \end{align*}
        \msol{x = -5}

    \section{Basic Physics Example}
        \begin{align*}
            \initialeq{p = mv}
            \variables{
                \varu{p}{\text{?}}{kg $ \frac{\text{m}}{\text{s}} $} \\
                \varu{m}{3.0}{k.g} \\
                \varu{v}{5.0}{m/s East}
            }
            \explain{Plug \& solve}{p = 3 \cdot 5}
        \end{align*}
        \msol{p = 15 \text{ kg} \cdot \frac{\text{m}}{\text{s}}}
